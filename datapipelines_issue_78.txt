SELECT *
FROM
(SELECT
  Contract_ID, Customer_ID AS Schedule_Customer_ID
FROM
  `bank-b.d6s_scoring.schedule_scoring`
WHERE
  Customer_ID NOT IN (
  SELECT
    DISTINCT Customer_ID
  FROM
    `bank-b.d6s_scoring.loan_scoring`)
GROUP BY 1, 2) 
LEFT JOIN
(SELECT Contract_ID, Customer_ID AS Loan_Customer_ID
FROM
  `bank-b.d6s_scoring.loan_scoring`)
USING(Contract_ID)
